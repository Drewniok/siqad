36  // attempt to read the keyed value from the Settings file. If the key is not
37  // found, it checks for default values. If a default value is found, the
38  // value is stored in the current settings file; otherwise, the app terminates
39  template<typename T>
40  T get(const QString &key)
41  {
42  QVariant var = DEFAULT_OVERRIDE ? defaults->value(key) : this->value(key);
43  T val;
44  // if key not found, get value from defaults
45  if (!var.isValid() && defaults != 0) {
46  var = defaults->value(key);
47 
48  // if key not in defaults, prompt and abort
49  if (!var.isValid()) { // terminate
50  qDebug() << tr("Searching for key: %1").arg(key);
51  qFatal(tr("Requested key missing in defaults... terminating").toLatin1().constData(),0);
52  } else {
53  val = var.value<T>();
54  }
55 
56  // save default value to current local settings
57  this->setValue(key, val);
58  } else if (defaults==0) { // terminate if no defaults
59  qFatal(tr("No default settings available... terminating").toLatin1().constData(),0);
60  } else if (var.isValid()) {
61  val = var.value<T>();
62  } else {
63  qFatal(tr("Unexpected key in Settings::get").toLatin1().constData(),0);
64  }
65 
66  return val;
67  }

74  static QString pathReplacement(QString path)
75  {
76  constructPathMap();
77 
78  // perform replacement
79  QRegExp regex("<(.*)?>");
80  while (path.indexOf(regex) != -1) {
81  if(!path_map.contains(regex.capturedTexts().first())) {
82  qFatal(tr("Path replacement failed, key '%1' not found.")
83  .arg(regex.capturedTexts().first()).toLatin1().constData(),0);
84  }
85  path.replace(regex, path_map[regex.capturedTexts().first()]);
86  }
87  return path;
88  }
